class Solution:
    def romanToInt(self, s: str) -> int:
        normal = {'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}
        abnormal = {'IV':4,'IX':9,'XL':40,'XC':90,'CD':400,'CM':900}
        i = 0
        res = 0
        while i < len(s):
            if i<len(s)-1 and normal[s[i]] < normal[s[i+1]]:
                res -= normal[s[i]]
            else:
                res += normal[s[i]]
            i += 1

        return res
        ## 时间复杂度O（N）
        ## 空间复杂度O（1）
